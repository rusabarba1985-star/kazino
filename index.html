import matplotlib.pyplot as plt import matplotlib.animation as animation import numpy as np # Настройка сцены: черный фон fig, ax = plt.subplots(figsize=(6, 4), facecolor='black') ax.set_facecolor('black') ax.set_xlim(0, 100) ax.set_ylim(0, 100) ax.axis('off') # Белый текст "Cartoon Game" (стиль Rubber Hose) text = ax.text(50, 50, "Cartoon Game", color='white', fontsize=40, fontweight='bold', fontname='serif', ha='center', va='center') # Создание "царапин" на пленке (серые полоски) num_lines = 5 lines = [ax.axhline(y=np.random.randint(0, 100), color='gray', alpha=0.3, linewidth=np.random.uniform(0.5, 2)) for _ in range(num_lines)] def update(frame): # Движение полосок сверху вниз for line in lines: new_y = line.get_ydata()[0] - 3 # Скорость падения if new_y < 0: new_y = 100 line.set_ydata([new_y, new_y]) line.set_alpha(np.random.uniform(0.1, 0.4)) # Мерцание # Эффект дрожания кадра (jitter) if frame % 2 == 0: text.set_position((50 + np.random.uniform(-0.3, 0.3), 50 + np.random.uniform(-0.3, 0.3))) return lines + [text] ani = animation.FuncAnimation(fig, update, frames=60, interval=50, blit=True) ani.save('rubber_hose_intro.gif', writer='pillow', fps=20) plt.show()
